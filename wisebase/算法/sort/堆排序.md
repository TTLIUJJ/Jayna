# 堆排序

- 原地排序
- 不稳定
- 时间复杂度
	- 建堆 O(N)
	- 调整堆 O(NlogN)
- 插入删除一个元素的时间复杂度 O(NlogN)
- 空间复杂度 O(1)


```java
import java.util.Arrays;
import java.util.Date;
import java.util.Random;

public class HeapSort {

    public static void heapSort(int []a){
        try {
            int N = a.length;
            for (int k = N / 2; k >= 0; --k) {
                sink(a, k, N);
            }
            
            while(N > 0){
                SortUtil.swap(a, 0, --N);
                sink(a, 0, N);
            }
        }catch (Exception e){
            e.printStackTrace();
        }
    }

    private static void sink(int []a, int k, int N){
        while(2*k < N){
            int j = 2 * k;
            if(j + 1 < N && a[j] < a[j+1]){
                ++j;
            }
            if(a[k] >= a[j]){
                break;
            }
            SortUtil.swap(a, k, j);
            k = j;
        }
    }

    public static void main(String []args){
        int []a = new int []{4, 2, 6, 1, 9, 5, 3, 8, 0, 7};
        HeapSort.heapSort(a);
        System.out.println(Arrays.toString(a));


        Random random = new Random(new Date().getTime());
        for(int i = 0; i < 10; ++i){
            int []b = new int[]{
                    random.nextInt(10),
                    random.nextInt(10),
                    random.nextInt(10),
                    random.nextInt(10),
                    random.nextInt(10),
                    random.nextInt(10),
                    random.nextInt(10),
                    random.nextInt(10),
            };
            System.out.print("before: " + Arrays.toString(b) + ", ");
            HeapSort.heapSort(b);
            System.out.println("after: " + Arrays.toString(b));
        }
    }
}
```
